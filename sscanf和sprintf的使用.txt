#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
struct S
{
	int n;
	float score;
	char arr[10];
};
int main() {
	struct S s = { 100,3.14f,"abcdef" };
	struct S tmp = { 0 };
	char buf[1024] = { 0 };

	//把格式化的数据转换成字符串储存在buf
	sprintf(buf, "%d %f %s", s.n, s.score, s.arr);
	//printf("%s\n", buf);
	//从buf中读取格式化的数据到tmp中
	sscanf(buf, "%d %f %s", &(tmp.n), &(tmp.score), tmp.arr);
	printf("%d %f %s\n", tmp.n,tmp.score,tmp.arr);
	return;
}
