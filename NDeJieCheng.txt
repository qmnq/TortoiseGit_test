#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int main() {
	int  i = 0;
	int sum = 0;//保存最后的结果
	int  n = 0;
	int ret = 1;//保存n地阶乘
	scanf("%d", &n);
	for (i = 1; i <= n; i++) {
		int j = 0;
		ret = 1;
		for (j = 1; j <= i; j++) {
			ret *= j;
		}
		sum += ret;
	}
	printf("%d\n", sum);
	return 0;
}
