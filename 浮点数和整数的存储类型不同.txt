#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int main() {
	int n = 9;
	//0000 0000 0000 0000 0000 0000 0000 1001――补码（十进制整型）
	//0 00000000 00000000000000000001001――float类型的补码
	//(-1)^0 * 00000000000000000001001 * 2^-126 (00000000的值是-126)
	float* pfloat = (float*)&n;
	printf("n的值是：%d\n", n);//9
	printf("pfloat的值是：%f\n", *pfloat);//0.000000

	*pfloat = 9.0;
	//1001.0  （9.0的二进制序列）
	//1.001 * 2^3
	//(-1)^0 *1.001 * 2^3   (2^3在内存中存储是3 + 127 =130)
	//0 10000010 00100000000000000000000――浮点数的存储
	//01000001000100000000000000000000――整型存储――
	printf("num的值是：%d\n", n);//1,091,567,616
	printf("pfloat的值是：%f\n", *pfloat);//9.0
	return 0;
}
//float的存储方式
// (-1)^S *   M   * 2^E
// (-1)^0 * 1.001 * 2^3
