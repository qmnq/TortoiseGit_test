#include <stdio.h>
bool isyear(int y){
    if((y%4==0&&y%100!=0)||y%400==0)
    return 1;
    return 0;
}
int main(){
    int y=1777;
    int m=4;
    int d=30;
    for(int i=0;i<8112;i++){
        d++;
        if(m==12&&d==32){
            y++;
            m=1;
            d=1;
            continue;
        }    
        if((m==1||m==3||m==5||m==7||m==8||m==10||m==12)&&d==32){
            m++;
            d=1;
            continue;
        }
        if((m==4||m==6||m==9||m==11)&&d==31){
            m++;
            d=1;
            continue;
        }
        if(m==2&&isyear(y)&&d==30){
            m++;
            d=1;
            continue;
        }
        if(m==2&&!isyear(y)&&d==29){
            m++;
            d=1;
            continue;
        }
    }
    printf("%d %d %d",y,m,d);
    return 0; 
}
